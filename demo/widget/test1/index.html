比较简陋，主要练习jQuery插件的写法,实际用途有限

            默认设置:
            defaults ={
                "imgbox":".scroll",//图片盒子,必填
                "navCount":".con",//导航盒子,必填
                "nextBtn":".next",//下一个,必填
                "prevBtn":".prev", //上一个,必填
                "on":"active", //导航标切换的类,必填
                "auto":true, //是否自动播放,选填
                "interval":3000, //图片轮换间隔，,选填
                "speed":500 //图片运动时间，,选填
            };
        
            // 图1使用默认参数
            $(".box").scroPic();

            // 图2 传入对应参数
            $(".box-2").scroPic({
            "imgbox":".scroll-2",
            "navCount":".con-2",
            "nextBtn":".next-2",
            "prevBtn":".prev-2",
            "on":"active"
            "interval":2000,
            "speed":200
        })
